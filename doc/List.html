def push(value)

	n = Node.new(value,@head,nil)
	if(@tail == nil)
		@tail = n
	end
	if(@head != nil)
		@head.prev = n
	end
	@head = n
end

def push_back(value)

	n = Node.new(value,nil,@tail)
	if(@head == nil)
		@head = n
	end
	if(@tail != nil)
		@tail.next = n
	end
	@tail = n
end
