def newRandDiet(pcarbs, pprots, plips)
		
	data =    []
	percents = []
	auxPer = 1.0
	i = 1 + (rand()*3)
	while i > 0 do
		thisauxPer = auxPer * rand()			
					
		#thiscarbs  = pcarbs * rand()
		#thispprots = pprots * rand()
		#thisplips  = plips  * rand()
		
		auxPer -= thisauxPer
		#pcarbs -= thiscarbs	*thisauxPer
		#pprots -= thispprots*thisauxPer		
		#plips  -= thisplips	*thisauxPer

		data 	<< newRandMenu(pcarbs,pprots,plips)
		percents << thisauxPer			

		i -= 1
	end
	data 	<< newRandMenu(pcarbs,pprots,plips)
	percents << auxPer

	p = Diet.new(
		rand().to_s,			
		data,
		percents
	)
	
end

def newRandPlate(pcarbs, pprots, plips)

	#pcarbs 	= opcarbs
	#pprots 	= opprots
	#plips	= oplips

	foods =    []
	percents = []
	prices =   []
	auxPer = 1.0
	auxPrice = 20
	i = 1 + (rand()*3)
	while i > 0 do
		thisauxPer = auxPer * rand()			
					
		#thiscarbs  = pcarbs * rand()
		#thispprots = pprots * rand()
		#thisplips  = plips  * rand()
		
		auxPer -= thisauxPer
		#pcarbs -= thiscarbs	*thisauxPer
		#pprots -= thispprots*thisauxPer		
		#plips  -= thisplips	*thisauxPer

		thisauxprice = auxPrice * rand()

		foods 	<< newRandFood(pcarbs,pprots,plips)
		percents << thisauxPer			
		prices << thisauxprice			

		i -= 1
	end
	foods 	<< newRandFood(pcarbs,pprots,plips)
	percents << auxPer
	prices << auxPrice * rand()


	p = Plate.new(
		rand().to_s,			
		foods,
		percents,
		prices
	)
	
end
